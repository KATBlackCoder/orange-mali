**Tech Stack:** React 19 + TypeScript + Vite, Tailwind CSS v4, shadcn/ui, Supabase (Auth + DB + RLS), Vercel, Vite PWA Plugin

## Task 11: Export CSV

**Files:**
- Create: `src/features/admin/ExportButton.tsx`

- [x] **Step 1: Créer le composant export**

```tsx
import { supabase } from '@/lib/supabase'
import { Button } from '@/components/ui/button'
import { Download } from 'lucide-react'

export function ExportButton({ formId }: { formId: string }) {
  async function handleExport() {
    // Récupérer soumissions + rows + values + fields
    const { data: fields } = await supabase.from('form_fields').select('*').eq('form_id', formId).order('ordre')
    const { data: submissions } = await supabase
      .from('submissions')
      .select('*, profiles(nom, telephone), submission_rows(*, row_values(*))')
      .eq('form_id', formId).eq('statut', 'valide')

    if (!fields || !submissions) return

    const headers = ['Employé', 'Téléphone', 'Date', 'Statut', ...fields.map(f => f.label)]
    const rows = submissions.flatMap(s =>
      ((s as any).submission_rows ?? []).map((row: any) => {
        const values = fields.map(f => {
          const v = row.row_values?.find((rv: any) => rv.field_id === f.id)
          return v?.valeur ?? ''
        })
        return [(s as any).profiles?.nom, (s as any).profiles?.telephone,
          new Date(s.created_at).toLocaleDateString('fr-ML'), s.statut, ...values]
      })
    )

    const csv = [headers, ...rows].map(r => r.join(';')).join('\n')
    const blob = new Blob(['\uFEFF' + csv], { type: 'text/csv;charset=utf-8' })
    const url = URL.createObjectURL(blob)
    const a = document.createElement('a'); a.href = url; a.download = 'export.csv'; a.click()
    URL.revokeObjectURL(url)
  }

  return (
    <Button variant="outline" onClick={handleExport}>
      <Download className="w-4 h-4 mr-2" /> Exporter CSV
    </Button>
  )
}
```
