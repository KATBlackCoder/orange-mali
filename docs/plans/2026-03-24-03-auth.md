## Task 4: Auth par téléphone + mot de passe

> **Conventions d'authentification :**
> - Email Supabase Auth (interne, jamais visible) : `telephone@orangemali.local`
> - Login affiché à l'utilisateur : `telephone@last_name.org`
> - Mot de passe par défaut : `ML` + telephone (ex: `ML70000000`)
> - `must_change_password = true` à la création → page changement mot de passe au 1er login
> - Seuls chef et sous_chef peuvent créer des comptes (via Edge Function, voir Chunk 6)

- [ ] **Step 2: Créer `src/lib/auth.ts`**

```ts
import { supabase } from './supabase'

// Convention: email interne = tel@orangemali.local
const toEmail = (tel: string) => `${tel.replace(/\s/g, '')}@orangemali.local`

export async function signIn(telephone: string, password: string) {
  return supabase.auth.signInWithPassword({
    email: toEmail(telephone),
    password,
  })
}

export async function signOut() {
  return supabase.auth.signOut()
}

export async function createUser(telephone: string, password: string, nom: string, role: string, zone?: string, parentId?: string) {
  // Appelé uniquement par un chef via Edge Function ou service role
  const { data, error } = await supabase.auth.admin.createUser({
    email: toEmail(telephone),
    password,
    email_confirm: true,
  })
  if (error || !data.user) throw error

  await supabase.from('profiles').insert({
    id: data.user.id,
    nom,
    telephone,
    role,
    zone,
    parent_id: parentId ?? null,
  })
  return data.user
}
```

- [ ] **Step 3: Créer `src/features/auth/useAuth.ts`**

```ts
import { useEffect, useState } from 'react'
import { supabase } from '@/lib/supabase'
import type { User } from '@supabase/supabase-js'

export function useAuth() {
  const [user, setUser] = useState<User | null>(null)
  const [loading, setLoading] = useState(true)

  useEffect(() => {
    supabase.auth.getSession().then(({ data }) => {
      setUser(data.session?.user ?? null)
      setLoading(false)
    })
    const { data: { subscription } } = supabase.auth.onAuthStateChange((_, session) => {
      setUser(session?.user ?? null)
    })
    return () => subscription.unsubscribe()
  }, [])

  return { user, loading }
}
```

- [ ] **Step 4: Créer `src/features/auth/LoginPage.tsx`**

```tsx
import { useState } from 'react'
import { signIn } from '@/lib/auth'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Label } from '@/components/ui/label'
import { Card, CardContent, CardHeader, CardTitle } from '@/components/ui/card'

export function LoginPage() {
  const [telephone, setTelephone] = useState('')
  const [password, setPassword] = useState('')
  const [error, setError] = useState('')
  const [loading, setLoading] = useState(false)

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault()
    setLoading(true)
    setError('')
    const { error } = await signIn(telephone, password)
    if (error) setError('Numéro ou mot de passe incorrect')
    setLoading(false)
  }

  return (
    <div className="min-h-screen flex items-center justify-center bg-orange-50 p-4">
      <Card className="w-full max-w-sm">
        <CardHeader>
          <CardTitle className="text-center text-orange-600">Orange Mali</CardTitle>
          <p className="text-center text-sm text-gray-500">Système de remontées</p>
        </CardHeader>
        <CardContent>
          <form onSubmit={handleSubmit} className="space-y-4">
            <div>
              <Label htmlFor="tel">Numéro de téléphone</Label>
              <Input id="tel" type="tel" placeholder="7X XX XX XX"
                value={telephone} onChange={e => setTelephone(e.target.value)} required />
            </div>
            <div>
              <Label htmlFor="pwd">Mot de passe</Label>
              <Input id="pwd" type="password"
                value={password} onChange={e => setPassword(e.target.value)} required />
            </div>
            {error && <p className="text-red-500 text-sm">{error}</p>}
            <Button type="submit" className="w-full bg-orange-500 hover:bg-orange-600" disabled={loading}>
              {loading ? 'Connexion...' : 'Se connecter'}
            </Button>
          </form>
        </CardContent>
      </Card>
    </div>
  )
}
```

- [ ] **Step 5: Mettre à jour `src/App.tsx` avec routing protégé**

```tsx
import { BrowserRouter, Routes, Route, Navigate } from 'react-router-dom'
import { useAuth } from '@/features/auth/useAuth'
import { LoginPage } from '@/features/auth/LoginPage'

function ProtectedRoute({ children }: { children: React.ReactNode }) {
  const { user, loading } = useAuth()
  if (loading) return <div className="flex items-center justify-center min-h-screen">Chargement...</div>
  if (!user) return <Navigate to="/login" replace />
  return <>{children}</>
}

export default function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/login" element={<LoginPage />} />
        <Route path="/*" element={
          <ProtectedRoute>
            <div>Dashboard (à venir)</div>
          </ProtectedRoute>
        } />
      </Routes>
    </BrowserRouter>
  )
}
```
